async function fetchPlace(btnData) {
    const placeData = await fetch(`https://geocoding-api.open-meteo.com/v1/search?name=${btnData}&count=10&language=en&format=json`);
    const responseData = await placeData.json();
    state.placeResults = responseData.results;
    console.log("Place-it:", state.placeResults)
    console.log("Button Data:", btnData)
    console.log("Place", responseData)
    return { responseData }
}
async function searchSuggestions(inputData) {
    const place = await fetchPlace(inputData);
    if (!place.responseData.results || place.responseData.results.length === 0) {
        console.log("No results found for:", inputData);
        state.placeResults = [];
        const dataList = document.querySelector("datalist");
        dataList.innerHTML = '';
        return;
    }

    state.placeResults = place.responseData.results;
    console.log("Search place data:", state.placeResults);
    const dataList = document.querySelector("#places");
    dataList.innerHTML = '';
    input.addEventListener("focus", () => {   
        console.log("Place Index-3:", state.placeIndex)
        dataList.style.opacity = 1;
    })
    place.responseData.results.forEach((result) => {
        const option = document.createElement("button");
        option.textContent = `${result.name}, ${result.admin1}, ${result.country}`;
        option.setAttribute("data-id", result.id);
        option.addEventListener("click", () => {
            option.setAttribute("data-selected", "");
            input.value = option.textContent;
            console.log("Place Index-2:", state.placeIndex)
            state.placeIndex = option.dataset.id;
            dataList.style.opacity = 0;
        })
        dataList.appendChild(option)
    });
}
async function fetchWeatherData(latitude, longitude) {
    const weatherData = await fetch(`https://api.open-meteo.com/v1/forecast?latitude=${latitude}&longitude=${longitude}&daily=weather_code,temperature_2m_max,temperature_2m_min,apparent_temperature_max,apparent_temperature_min,sunrise,sunset&hourly=temperature_2m,relative_humidity_2m,precipitation,rain,weather_code,wind_speed_10m,visibility&current=temperature_2m,relative_humidity_2m,is_day,wind_speed_10m,apparent_temperature,precipitation,rain,showers,cloud_cover,weather_code&timezone=auto`);
    const responseData = await weatherData.json();
    weatherDetails.temperature = responseData.current.temperature_2m;
    weatherDetails.feelsLike = responseData.current.apparent_temperature;
    weatherDetails.humidity = responseData.current.relative_humidity_2m;
    weatherDetails.rain = responseData.current.rain;
    const trueVisibility = responseData.hourly.visibility[0] / 1000;
    weatherDetails.visibility = trueVisibility.toString() + "km";
    weatherDetails.weatherCode = responseData.current.weather_code;
    weatherDetails.day = responseData.current.time.slice(0, 10);
    weatherDetails.time = responseData.current.time.slice(11);
    weatherDetails.wind = responseData.current.wind_speed_10m;
    console.log("Weather Data", responseData);
    console.log("Rain:", responseData.current.rain);
    console.log("Current temp:", responseData.current.temperature_2m);
    console.log("Visibility:", trueVisibility.toString().slice(0,4) + "km")
}
async function fetchSpecificPlace(id) {
    const place = await fetch(`https://geocoding-api.open-meteo.com/v1/get?id=${id}`);
    const response = await place.json();
    weatherDetails.city = response.name;
    weatherDetails.country = response.country;
    console.log("Specific Place:", response);
    console.log("Index-2:", id)
    return response;
}
